MySQL 入门使用篇(1)

2023/7/30 DataBase MySQL

从建库到删库再到跑路

安装就不多记录了,不同系统不同方式,一般就两种

  1. 通过系统的软件包管理工具进行一键安装
  2. 官网/镜像网站下载安装包手动安装

数据库基本操作

  1. 创建库
    先看看创建数据库的语法格式,然后创建一个数据库test;
CREATE DATABASE [IF NOT EXISTS] <数据库名>
[[DEFAULT] CHARACTER SET <字符集名>]
[[DEFAULT] COLLATE <校对规则名>];

CREATE DATABASE test;
  1. 查看数据库
    语法格式,然后输入查看下;
SHOW DATABASES [LIKE '数据库名'];

show databases;
  1. 修改数据库
    修改数据库的语法格式,参数可选;尝试将之前创建的test数据库的字符集改为utf8;
ALTER DATABASE [数据库名] {
[ DEFAULT ] CHARACTER SET <字符集名> |
[ DEFAULT ] COLLATE <校对规则名>}

ALTER DATABASE test CHARACTER SET utf8;
  1. 选择数据库
    use进行选择,之后表的操作都在选择的数据库下.
USE <数据库名>

use test;
  1. 删除数据库
DROP DATABASE [ IF EXISTS ] <数据库名>

DROP DATABASE test;

数据表基本操作

  1. 创建数据表
CREATE TABLE <表名> (
  <列名1> <类型1> ,
        .
        .
        .
  <列名n> <类型n>
)[表选项][分区选项];

use test;
CREATE TABLE user(
  uid INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(50) NOT NULL,
  age INT
)
  1. 修改数据表
ALTER TABLE <表名>  { ADD COLUMN <列名> <类型>
| CHANGE COLUMN <旧列名> <新列名> <新列类型>
| ALTER COLUMN <列名> { SET DEFAULT <默认值> | DROP DEFAULT }
| MODIFY COLUMN <列名> <类型>
| DROP COLUMN <列名>
| RENAME TO <新表名>
| CHARACTER SET <字符集名>
| COLLATE <校对规则名> }

ALTER TABLE user ADD COLUMN email VARCHAR(100);
  1. 查看数据表的结构
DESCRIBE <表名>;

DESCRIBE user;
  1. 删除数据表
DROP TABLE [IF EXISTS] 表名1 [ ,表名2, 表名3 ...];

DROP TABLE user;

表内的数据进行增删改查

  1. 增
INSERT INTO <表名> [ <列名1> [ , … <列名n>] ] VALUES (值1) [… , (值n) ];

INSERT INTO user (name, age) VALUES ("coky", 24);
  1. 查
SELECT {* | <字段列名>}
[
FROM <表 1>, <表 2>…
[WHERE <表达式>
[GROUP BY <group by definition>
[HAVING <expression> [{<operator> <expression>}…]]
[ORDER BY <order by definition>]
[LIMIT[<offset>,] <row count>]
]

SELECT * FROM user;
  1. 改
UPDATE <表名> SET 字段 1=值 1 [,字段 2=值 2… ] [WHERE 子句 ]
[ORDER BY 子句] [LIMIT 子句]

UPDATE user SET age = 25 WHERE uid = 1;
  1. 删
    删除数据注意条件,没条件就是全删.
DELETE FROM <表名> [WHERE 子句] [ORDER BY 子句] [LIMIT 子句]

DELETE FROM user WHERE uid = 1;

退出

quit;
